```C# 
// Class Encapsulation
using System;

namespace BasicClasses
{
  class Forest
  {
    public int age;
    private string biome; //field
    
    //Automatic property
    public string Name
    { get; set; }
    
    public int Trees
    { get; set; }
    
    public string Biome //property
    {
      get { return biome; }
      set
      {
        if (value == "Tropical" ||
            value == "Temperate" ||
            value == "Boreal")
        {
          biome = value;
        }
        else
        {
          biome = "Unknown";
        }
      }
    }
  }

}
```
