```C#
using System;

namespace BasicClasses
{
  class Forest
  {

    public int Grow()   //defined method that manipulates the Age and Trees fields
   {
     Trees = Trees + 30;
     Age = Age + 1;
     return Trees;
   }

       public int Burn()
   {
     Trees = Trees - 20;
     Age = Age + 1;
     return Trees;
   }

    public int age;
    private string biome;
    
    public string Name
    { get; set; }
    
    public int Trees
    { get; set; }
    
    public string Biome
    {
      get { return biome; }
      set
      {
        if (value == "Tropical" ||
            value == "Temperate" ||
            value == "Boreal")
        {
          biome = value;
        }
        else
        {
          biome = "Unknown";
        }
      }
    }
    
    public int Age
    { 
      get { return age; }
      private set { age = value; }
    }
    
  }

}
```
